### 深度优先搜索
严蔚敏阿姨的类C语言版本如下
```
void DFS(Graph G, int v){
    visited[v] = true;
    visit(v);//访问节点
    //FirstAdjVex: 获取与当前节点相连的第一个节点
    //NextAdjVex: 获取与当前节点相连的下一个节点
    for(w = FirstAdjVex(G, v); w != 0; w = NextAdjVex(G, v, w)){
        if(!visited[w]){
            DFS(G, w)
        }
    }
}
void DFSTraverse(Graph G){
    for(v = 0; v < G.vexnum; v++){
        if(!visited[v]){
            DFS(G, v);
        }
    }
}
```
如果使用邻接矩阵存储图需要O(n^2)的空间复杂度，如果图很大，数组可能开不下。而且在稀疏图中，浪费非常严重。可以使用vector数组保存图
```c++
vector<int>G[maxn];
void read(){
    int u,v,m;//边的起点，边的终点，边的数目
    scanf("%d",&n);
    for(int i=0;i<m;i++){
        scanf("%d%d",&u,&v);
        G[u].push_back(v);
        G[v].push_back(u);
    }
}
void dfs(int u){
    visited[u]=true;
    int d=G[u].size();//顶点u作为起点的边的数目
    for(int i=0;i<d;i++){
        int v=G[u][i];//u的第i条边的终点
        if(!visited[v]){
            dfs(v);
        }
    }
}
```

### 广度优先搜索
```
void BSF(Graph G,int v){
    visited[v]=true;
    visit(v);
    Enqueue(Q,v);
    while(!QueueEmpty(Q)){
        Dequeue(Q,u);
        for(w=FirstAdjVex(G,v);w!=0;w=NextAdjVex(G,v,w)){
            if(!visited[w]){
                visited[w]=true;
                Enqueue(Q,w);
            }//if
        }//for
    }//while
}
void BFSTraverse(Graph G){
    InitQueue(Q);
    for(v=0;v<G.vexnum;v++){
        if(!visited[v]){
            BFS(G,v);
        }
    }
}
```

### Kruskal
伪代码如下
```
把所有边排序,记最小的边为e[i]
for(int i=0;i<m;i++){
    if(e[i].u和e[i].v不在同一个联通分量){
        把e[i]加入MST（最小生成树）
        合并e[i].u和e[i].v所在联通分量
    }
}
```
可以使用并查集表示联通分量。
```
int find(int x) {
	return p[x] == x ? x : p[x] = find(p[x]);
}
```
有关并查集可以参考hihocoder 1066
```c++
#include<iostream>
#include<string>
#include<map>
using namespace std;
const int maxn = 1e5+10;

map<string, int>z;
int pre[maxn];

int find(int x) {
	return pre[x] == x ? x : pre[x] = find(pre[x]);
}
int main()
{
	int n, temp = 0;
	int idx1, idx2;
	cin >> n;
	bool op;
	string s1, s2;
	for (int i = 1; i <= n; i++) {
		pre[i] = i;
	}
	while (n--) {
		cin >> op >> s1 >> s2;
		idx1 = z[s1];
		idx2 = z[s2];
		if (!idx1) {
			idx1 = z[s1] = ++temp;
		}
		if (!idx2) {
			idx2 = z[s2] = ++temp;
		}
		if (!op) {
			pre[find(idx1)] = find(idx2);
		}
		else {
			if (find(idx1) == find(idx2)) {
				cout << "yes" << endl;
			}
			else {
				cout << "no" << endl;
			}
		}
	}
	return 0;
}
```
如果p[x]==x，说明x是树根，返回x，否则返回x的父节点p[x]所在的树根。
假设第i小的边的端点和权值保存在u[i],v[i],w[i]中，以hihocoder 1098为例
```c++
#include<iostream>
#include<algorithm>
using namespace std;
const int maxm = 1000010;
const int maxn = 100010;
int m, n;
int u[maxm];
int v[maxm];
int w[maxm];
int p[maxn];
int r[maxm];
bool cmp(const int i, const int j) { return w[i] < w[j]; }
int find(int x) {
	return p[x] == x ? x : p[x] = find(p[x]);
}
void Kruskal() {
	int ans = 0;
	for (int i = 0; i < n; i++) {//初始化并查集
		p[i] = i;
	}
	for (int i = 0; i < m; i++) {//初始化边编号
		r[i] = i;
	}
	sort(r, r + m, cmp);
	for (int i = 0; i < m; i++) {
		int e = r[i]; int x = find(u[e]); int y = find(v[e]);
		if (x != y) {
			ans += w[e];
			p[x] = y;
		}
	}
	cout << ans << endl;
}
int main()
{
	cin >> n >> m;
	for (int i = 0; i < m; i++) {
		cin >> u[i] >> v[i] >> w[i];
	}
	Kruskal();
}
```
